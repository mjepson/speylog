// Speylog Edge Function: fetch-conditions // Henter vanndata (NVE HydAPI) og lufttemperatur (MET Frost) for en elv + tidspunkt. // Kalles fra appen: sb.functions.invoke('fetch-conditions', { body: { river, timestamp } }) // // Secrets (settes i Dashboard -> Edge Functions -> Secrets): // NVE_API_KEY - fra https://hydapi.nve.no (gratis registrering) // FROST_CLIENT_ID - fra https://frost.met.no/auth/requestCredentials.html import { createClient } from "jsr:@supabase/supabase-js@2"; const CORS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", }; const HYDAPI = "https://hydapi.nve.no/api/v1"; // NVE parameter-IDer: 1000 = vannstand (m), 1001 = vannføring (m3/s), 1003 = vanntemperatur (°C) const NVE_PARAMS = "1000,1001,1003"; function isoWindow(ts: Date, hours = 2) { const from = new Date(ts.getTime() - hours * 3600_000); const to = new Date(ts.getTime() + hours * 3600_000); return `${from.toISOString()}/${to.toISOString()}`; } // Velg observasjonen nærmest fangsttidspunktet function closest(observations: { time: string; value: number | null }[], ts: Date) { let best: number | null = null; let bestDiff = Infinity; for (const o of observations ?? []) { if (o.value == null) continue; const diff = Math.abs(new Date(o.time).getTime() - ts.getTime()); if (diff < bestDiff) { bestDiff = diff; best = o.value; } } return best; } Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: CORS }); try { const { river, timestamp } = await req.json(); if (!river || !timestamp) { return json({ error: "river og timestamp er påkrevd" }, 400); } const ts = new Date(timestamp); if (isNaN(ts.getTime())) return json({ error: "ugyldig timestamp" }, 400); const nveKey = Deno.env.get("NVE_API_KEY"); const frostId = Deno.env.get("FROST_CLIENT_ID"); if (!nveKey) return json({ error: "NVE_API_KEY mangler i secrets" }, 500); // 1. Slå opp stasjon for elven const sb = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!, ); const { data: station, error: stErr } = await sb .from("river_stations") .select("*") .eq("river_name", river) .maybeSingle(); if (stErr) return json({ error: stErr.message }, 500); if (!station) return json({ error: `Ingen NVE-stasjon registrert for ${river}`, no_station: true }, 404); const nveHeaders = { "X-API-Key": nveKey, Accept: "application/json" }; // 2. Hent stasjonskoordinater hvis vi ikke har dem (trengs for Frost) let { latitude, longitude } = station; if (latitude == null || longitude == null) { const r = await fetch(`${HYDAPI}/Stations?StationId=${station.nve_station_id}`, { headers: nveHeaders }); if (r.ok) { const meta = await r.json(); const s = meta?.data?.[0]; if (s?.latitude != null) { latitude = s.latitude; longitude = s.longitude; await sb.from("river_stations") .update({ latitude, longitude }) .eq("river_name", river); } } } // 3. NVE-observasjoner rundt fangsttidspunktet const obsUrl = `${HYDAPI}/Observations?StationId=${station.nve_station_id}` + `&Parameter=${NVE_PARAMS}&ResolutionTime=60&ReferenceTime=${isoWindow(ts)}`; const obsRes = await fetch(obsUrl, { headers: nveHeaders }); if (!obsRes.ok) return json({ error: `NVE-feil: ${obsRes.status}` }, 502); const obs = await obsRes.json(); const byParam: Record = {}; for (const series of obs?.data ?? []) { byParam[series.parameter] = closest(series.observations, ts); } // 4. Lufttemperatur fra Frost (nærmeste værstasjon) let airTemp: number | null = null; let weatherStation: string | null = null; if (frostId && latitude != null && longitude != null) { const auth = "Basic " + btoa(`${frostId}:`); try { const srcRes = await fetch( `https://frost.met.no/sources/v0.jsonld?types=SensorSystem&elements=air_temperature` + `&geometry=nearest(POINT(${longitude} ${latitude}))`, { headers: { Authorization: auth } }, ); if (srcRes.ok) { const src = await srcRes.json(); const source = src?.data?.[0]; if (source?.id) { weatherStation = source.name ?? source.id; const oRes = await fetch( `https://frost.met.no/observations/v0.jsonld?sources=${source.id}` + `&referencetime=${isoWindow(ts)}&elements=air_temperature`, { headers: { Authorization: auth } }, ); if (oRes.ok) { const o = await oRes.json(); const flat = (o?.data ?? []).map((d: any) => ({ time: d.referenceTime, value: d.observations?.[0]?.value ?? null, })); airTemp = closest(flat, ts); } } } } catch (_) { /* Frost nede -> fortsett uten lufttemp */ } } return json({ water_stage: byParam[1000] ?? null, // m water_flow: byParam[1001] ?? null, // m3/s water_temp: byParam[1003] ?? null, // °C air_temp: airTemp, // °C nve_station: station.nve_station_name ?? station.nve_station_id, weather_station: weatherStation, fetched_at: new Date().toISOString(), }); } catch (e) { return json({ error: String(e) }, 500); } }); function json(body: unknown, status = 200) { return new Response(JSON.stringify(body), { status, headers: { ...CORS, "Content-Type": "application/json" }, }); }
